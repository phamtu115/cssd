  <script>
    let allData = [];
    let lastUpdateValue = "";

    function normalizeStr(str) {
      return str.toLowerCase()
        .normalize("NFD")
        .replace(/[\u0300-\u036f]/g, "")
        .trim();
    }

    function getRowClass(status) {
      if (!status) return "";
      const s = normalizeStr(status);
      if (s.includes("may loi")) return "status-error";
      if (s.includes("cho dong goi")) return "status-waiting";
      if (s.includes("hoan thanh tiet khuan")) return "status-sterile";
      if (s.includes("van chuyen")) return "status-shipped";
      if (s.includes("dang su dung")) return "status-using"; 
      if (s.includes("dang tiet khuan")) return "status-sterilizing";
      return "";
    }

    function loadData() {
      google.script.run.withSuccessHandler(render).getData();
    }

    function render(data) {
      data.sort((a, b) => {
        const dateA = new Date(`${a.ngay}T${a.giodukienhoanthanh}`);
        const dateB = new Date(`${b.ngay}T${b.giodukienhoanthanh}`);
        return dateB - dateA;
      });

      allData = data;

      const statusSelect = document.getElementById("filter");
      statusSelect.innerHTML = "";

      let uniqueStatuses = Array.from(new Set(data.map(r => r.trangthai).filter(x => x)));
      uniqueStatuses.sort((a, b) => normalizeStr(a).localeCompare(normalizeStr(b)));

      statusSelect.innerHTML = `<option value="all">-- Tất cả trạng thái --</option>`;
      uniqueStatuses.forEach(val => {
        statusSelect.innerHTML += `<option value="${normalizeStr(val)}">${val}</option>`;
      });

      google.script.run.withSuccessHandler(function(V2Date) {
        const timeSelect = document.getElementById("timeFilter");
        timeSelect.innerHTML = "";

        let timeSet = [...new Set(data.map(r => r.ngay).filter(x => x))];
        if (!timeSet.includes(V2Date)) timeSet.push(V2Date);

        timeSet.sort((a, b) => {
          const [da, ma, ya] = a.split("/").map(Number);
          const [db, mb, yb] = b.split("/").map(Number);
          return new Date(yb, mb - 1, db) - new Date(ya, ma - 1, da);
        });

        timeSelect.innerHTML = `<option value="all">-- Tất cả ngày --</option>`;
        timeSet.forEach(val => {
          timeSelect.innerHTML += `<option value="${val}">${val}</option>`;
        });

        showData();
      }).getLastUpdate();
    }

    function showData() {
      const statusFilter = normalizeStr(document.getElementById("filter").value);
      const timeFilter = document.getElementById("timeFilter").value;
      const searchTenBo = normalizeStr(document.getElementById("searchTenBo").value);

      const tbody = document.getElementById("tbody");
      tbody.innerHTML = "";

      let filtered = allData.filter(r => {
        const rStatus = normalizeStr(r.trangthai || "");
        const rTenBo = normalizeStr(r.tenbo || "");

        return (statusFilter === "all" || rStatus === statusFilter)
          && (timeFilter === "all" || r.ngay === timeFilter)
          && (searchTenBo === "" || rTenBo.includes(searchTenBo));
      });

      filtered.forEach(row => {
        const tr = document.createElement("tr");
        tr.className = getRowClass(row.trangthai);

        tr.innerHTML = `
          <td>
            <select onchange="updateStatus('${row.id}', this.value)">
              <option value="dang tiet khuan" ${normalizeStr(row.trangthai) === "dang tiet khuan" ? "selected" : ""}>Đang tiệt khuẩn</option>
              <option value="hoan thanh tiet khuan" ${normalizeStr(row.trangthai) === "hoan thanh tiet khuan" ? "selected" : ""}>Hoàn thành tiệt khuẩn</option>
              <option value="dang su dung" ${normalizeStr(row.trangthai) === "dang su dung" ? "selected" : ""}>Đang sử dụng</option>
              <option value="may loi" ${normalizeStr(row.trangthai) === "may loi" ? "selected" : ""}>Máy lỗi</option>
              <option value="cho dong goi" ${normalizeStr(row.trangthai) === "cho dong goi" ? "selected" : ""}>Chờ đóng</option>
              <option value="van chuyen" ${normalizeStr(row.trangthai) === "van chuyen" ? "selected" : ""}>Đã vận chuyển</option>
            </select>
          </td>
          <td>${row.tenbo}</td>
          <td>${row.donle}</td>
          <td>${row.thuthuat}</td>
          <td>${row.giodukienhoanthanh}</td>
        `;
        tbody.appendChild(tr);
      });
    }

    function updateStatus(id, value) {
      google.script.run.withSuccessHandler(loadData).updateStatus(id, value);
    }

    window.onload = () => {
      google.script.run.withSuccessHandler(function(v) {
        lastUpdateValue = v;
        loadData();
      }).getLastUpdate();
    };
  </script>
